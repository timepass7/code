# CREATE A DATAFRAME  
students<-data.frame(Name=c("Chris","Steve","Friday","Vinson", 
"Mark","Max","Hell","Ken"),Marks=c(91,87,95,81,84,78,94,90)) 

# BAR CHART FROM DATAFRAME  
barplot(students$Marks,names.arg=students$Name,col="orange", 
main="Student Marks - Bar Chart",xlab ="Students",ylab="Marks")  

# PIE CHART FROM DATAFRAME  
pie(students$Marks,labels=paste(students$Name,"-", 
students$Marks),col=rainbow(5),main="Student Marks-Pie Chart")  

# SAVE DATAFRAME TO CSV  
write.csv(students,"Students.csv",row.names=FALSE)
